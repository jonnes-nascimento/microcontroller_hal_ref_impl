162void Gpio_Init(const GpioConfig_t * const Config)
163{
164 printf("\r\nInitializing GPIO...\r\n");
165
166 /* This is for test only - this code must be moved */
167 //*SystemOscilator = 0x70; /**< Switch to 8 MHz system clock */
168 //*SystemComparator = 0x07; /**< Comparators off */
169 //*SystemADConverters = 0x00; /**< AD Converters off */
170
171 for (uint8_t port = 0; port < GPIO_PORT_COUNT; port++)
172 {
173 //*GpioPort[port] = 0x00; /**< Resets all GPIO pins */
174
175 /* Sets the GPIOs */
176 for (uint8_t i = 0; i < GPIO_CHANNEL_COUNT; i++)
177 {
178 // *GpioDirection[port] = (Config[i].Direction << i);
179 trisio |= (Config[i].Direction << i);
180
181 // *GpioPort[port] = (Config[i].State << i);
182 gpio |= (Config[i].State << i);
183
184 switch (Config[i].Function)
185 {
186 case GPIO_PIN_FUNCTION_ANALOG:
187 if (Config[i].Direction == GPIO_PIN_DIRECTION_INPUT)
188 {
189 //*GpioFunction[port] |= (Config[i].Direction << i);
190 ansel |= (Config[i].Direction << i);
191 }
192 else
193 {
194 //*GpioFunction[port] |= (HIGH << i);
195 ansel &= ~(HIGH << i);
196 }
197 break;
198 case GPIO_PIN_FUNCTION_EXTERNAL_INTERRUPT:
199
200 // Enable the external interrupt on GP2/INT
201
202 // Sets the edge trigger of the interrupt
203 if (Config[i].InterruptEdge == GPIO_INTERRUPT_EDGE_FALLING)
204 {
205 //*GpioOption[port] &= ~INTEDG_BIT_MASK;
206 option &= ~INTEDG_BIT_MASK;
207 }
208 else if (Config[i].InterruptEdge == GPIO_INTERRUPT_EDGE_RISING)
209 {
210 //*GpioOption[port] |= INTEDG_BIT_MASK;
211 option |= INTEDG_BIT_MASK;
212 }
213 else
214 {
215 // do nothing
216 }
217
218 // Clear the external interrupt flag INTF
219 //*GpioInterruptControl[port] &= ~INTF_BIT_MASK;
220 intcon &= ~INTF_BIT_MASK;
221
222 // Enable external interrupts
223 //*GpioInterruptControl[port] |= INTE_BIT_MASK;
224 intcon |= INTE_BIT_MASK;
225
226 // Enable global interrupts
227 //*GpioInterruptControl[port] |= GIE1_BIT_MASK;
228 intcon |= GIE1_BIT_MASK;
229
230 // Save the callback function executor address to the FSR register
231 //*FSR = &Gpio_Interrupt_Callback_Executor;
232 fsr = Gpio_Interrupt_Callback_Executor;
233
234 break;
235 } // switch Config[i].Function
236 } // for GPIO_CHANNEL_COUNT
237 } // for GPIO_PORT_COUNT
238
239 printf("OPTION register: "BYTE_TO_BINARY_PATTERN"\n", BYTE_TO_BINARY(option));
240 printf("INTCON register: "BYTE_TO_BINARY_PATTERN"\n", BYTE_TO_BINARY(intcon));
241 printf("TRISIO register: "BYTE_TO_BINARY_PATTERN"\n", BYTE_TO_BINARY(trisio));
242 printf("ANSEL register: "BYTE_TO_BINARY_PATTERN"\n", BYTE_TO_BINARY(ansel));
243 printf("GPIO register: "BYTE_TO_BINARY_PATTERN"\n", BYTE_TO_BINARY(gpio));
244 printf("FSR register: %x\r\n", &(*fsr));
245 fsr();
246}
